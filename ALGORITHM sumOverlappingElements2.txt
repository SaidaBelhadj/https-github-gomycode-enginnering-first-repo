ALGORITHM  sumOverlappingElements2 

  VAR 
       i , j: INTEGER;
       sum  : INTEGER :=0;
       set1 : ARRAY_OF INTEGER;// array set1 
       set2 : ARRAY_OF INTEGER;// array set2
   BEGIN
      Read(set1);
      Read(set2);

   
        FOR i FROM 0 TO set1.length-1 STEP 1 DO
          FOR j FROM 0 TO set2.length-1 STEP 1 DO
            
            IF (set1[i] = set2[j]) 
               sum := sum + set1[i];
               break;

            END_IF
           END_FOR 
        END_FOR
      FOR j FROM 0 TO set2.length-1 STEP 1 DO
          FOR i FROM 0 TO set1.length-1 STEP 1 DO
            
            IF (set1[i]= set2[j]) 
               sum := sum + set2[j];
               break;
               
            END_IF
           END_FOR 
        END_FOR
    

   Write("Overlapping Elements Sum : " + sum);

 END
